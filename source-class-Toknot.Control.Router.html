110:     /**
111:      * Set Controler info that under application, if CLI mode, will set request method is CLI
112:      * 
113:      */
114:     public function routerRule() {
115:         if ($this->routerMode == self::ROUTER_GET_QUERY) {
116:             if (empty($_GET['c'])) {
117:                 $this->spacePath = $this->defaultClass;
118:             } else {
119:                 $this->spacePath = '\\' . strtr($_GET['c'], '.', '\\');
120:             }
121:         } else {
122:             if (PHP_SAPI == 'cli') {
123:                 if (isset($_SERVER['argv'][1])) {
124:                     $_SERVER['REQUEST_URI'] = $_SERVER['argv'][1];
125:                 } else {
126:                     $_SERVER['REQUEST_URI'] = '/';
127:                 }
128:             }
129:             if (($pos = strpos($_SERVER['REQUEST_URI'], '?')) !== false) {
130:                 $urlPath = substr($_SERVER['REQUEST_URI'], 0, $pos);
131:             } else {
132:                 $urlPath = $_SERVER['REQUEST_URI'];
133:             }
134:             $spacePath = strtr($urlPath, '/', StandardAutoloader::NS_SEPARATOR);
135:             $spacePath = $spacePath == StandardAutoloader::NS_SEPARATOR ? $this->defaultClass : $spacePath;
136:             if ($this->routerDepth > 0) {
137:                 $name = strtok($spacePath, StandardAutoloader::NS_SEPARATOR);
138:                 $this->spacePath = '';
139:                 $depth = 0;
140:                 while ($name) {
141:                     if ($depth <= $this->routerDepth) {
142:                         $this->spacePath .= StandardAutoloader::NS_SEPARATOR . ucfirst($name);
143:                     } else {
144:                         $this->suffixPart[] = $name;
145:                     }
146:                     $name = strtok(StandardAutoloader::NS_SEPARATOR);
147:                     $depth++;
148:                 }
149:             } else {
150:                 $this->spacePath = $spacePath;
151:             }
152:         }
153:     }

183:     /**
184:      * Invoke Application Controller, the method will call application of Controller what is
185:      * $this->routerNameSpace\Controller{$this->spacePath}, and router action by request method
186:      * 
187:      * @param \Toknot\Control\FMAI $appContext
188:      * @throws BadClassCallException
189:      * @throws StandardException
190:      */
191:     public function invoke(FMAI $FMAI) {
192:         $method = $this->getRequestMethod();
193:         $invokeClass = "{$this->routerNameSpace}\Controller{$this->spacePath}";
194:         $classFile = StandardAutoloader::transformClassNameToFilename($invokeClass, $this->routerPath);
195:         $classExist = false;
196:         $caseClassFile = FileObject::fileExistCase($classFile);
197:         if ($caseClassFile) {
198:             include_once $caseClassFile;
199:             $invokeClass = str_replace($this->routerPath, '', $caseClassFile);
200:             $invokeClass = strtr($invokeClass, '/', '\\');
201:             $invokeClass = $this->routerNameSpace . strtok($invokeClass, '.');
202:             $classExist = class_exists($invokeClass, false);
203:         }
204:         if (!$classExist) {
205:             $dir = StandardAutoloader::transformClassNameToFilename($invokeClass, $this->routerPath);
206:             if (is_dir($dir) && $this->defaultClass != null) {
207:                 $invokeClass = "{$this->routerNameSpace}\Controller{$this->spacePath}\{$this->defaultClass}";
208:                 $classFile = StandardAutoloader::transformClassNameToFilename($invokeClass, $this->routerPath);
209:                 if (is_file($classFile)) {
210:                     include $classFile;
211:                 }
212:                 if (!class_exists($invokeClass, false)) {
213:                     $invokeClass = $this->invokeNotFoundController();
214:                 }
215:             } else {
216:                 $this->invokeNotFoundController($invokeClass);
217:             }
218:         }
219:         $invokeClassReflection = new ReflectionClass($invokeClass);
220: 
221:         $FMAI->setURIOutRouterPath($this->suffixPart, $method);
222: 
223:         $invokeObject = $invokeClassReflection->newInstance($FMAI);
224:         if (!$invokeClassReflection->hasMethod($method)) {
225:             if (DEVELOPMENT) {
226:                 throw new StandardException("Not Support Request Method ($method)");
227:             } else {
228:                 header('405 Method Not Allowed');
229:                 if ($this->methodNotAllowedController === null) {
230:                     $invokeClass = "{$this->routerNameSpace}\{$this->methodNotAllowedController}";
231:                     $classFile = StandardAutoloader::transformClassNameToFilename($invokeClass, $this->routerPath);
232:                     if (is_file($classFile)) {
233:                         include $classFile;
234:                     }
235:                     if (!class_exists($invokeClass, false)) {
236:                         die('405 Method Not Allowed');
237:                     }
238:                 } else {
239:                     die('405 Method Not Allowed');
240:                 }
241:             }
242:         }
243:         $stat = $FMAI->invokeBefore($invokeClassReflection);
244:         if ($stat === true) {
245:             $invokeObject->$method();
246:         }
247:         $FMAI->invokeAfter();
248:     }
